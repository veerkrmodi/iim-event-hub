import React from 'react';
import { Calendar, Clock, MapPin, Bell, Download, Edit2, Trash2 } from 'lucide-react';
import { getGoogleCalendarLink, downloadICS } from '../utils/calendarUtils';

const EventCard = ({ 
  event, 
  view, 
  user, 
  onEdit, 
  onDelete 
}) => {
  const canEditDelete = view === 'admin' || (view === 'club' && event.createdBy === user?.uid);

  return (
    <article className="bg-white p-6 rounded-2xl shadow-sm border border-slate-100 hover:border-indigo-200 hover:shadow-md transition-all group">
      <div className="flex flex-col md:flex-row justify-between gap-6">
        <div className="flex-1">
          <div className="flex items-center gap-3 mb-3">
            <span className="bg-indigo-50 text-indigo-600 text-[10px] font-black px-2.5 py-1 rounded-md uppercase tracking-tighter">
              {event.category}
            </span>
            <span className="text-xs font-semibold text-slate-400 tracking-tight">{event.club}</span>
          </div>
          <h3 className="text-xl font-extrabold mb-3 group-hover:text-indigo-600 transition-colors leading-tight">
            {event.title}
          </h3>
          <div className="grid grid-cols-1 sm:grid-cols-2 gap-x-6 gap-y-2.5 text-sm text-slate-600 mb-5">
            <div className="flex items-center gap-2.5">
              <Calendar size={16} className="text-indigo-400" />
              {new Date(event.date).toLocaleDateString('en-GB', { 
                day: 'numeric', 
                month: 'long', 
                year: 'numeric' 
              })}
            </div>
            <div className="flex items-center gap-2.5">
              <Clock size={16} className="text-indigo-400" />
              {event.time}
            </div>
            <div className="flex items-center gap-2.5 sm:col-span-2">
              <MapPin size={16} className="text-indigo-400" />
              <span className="truncate">{event.venue}</span>
            </div>
          </div>
          <p className="text-sm text-slate-500 leading-relaxed line-clamp-2">
            {event.description}
          </p>
        </div>

        <div className="flex flex-row md:flex-col gap-2 justify-end items-center md:items-stretch">
          <a 
            href={getGoogleCalendarLink(event)}
            target="_blank" 
            rel="noopener noreferrer"
            className="flex-1 md:flex-none text-center bg-slate-50 hover:bg-indigo-50 border border-slate-100 p-2.5 rounded-xl text-xs font-bold flex items-center justify-center gap-2 transition-all"
          >
            <Bell size={14} className="text-indigo-500" /> Reminder
          </a>
          <button 
            onClick={() => downloadICS(event)}
            className="flex-1 md:flex-none text-center bg-slate-50 hover:bg-emerald-50 border border-slate-100 p-2.5 rounded-xl text-xs font-bold flex items-center justify-center gap-2 transition-all"
          >
            <Download size={14} className="text-emerald-500" /> Save Offline
          </button>
          
          {canEditDelete && (
            <div className="flex gap-2 pt-3 mt-1 border-t border-slate-50 md:border-none">
              <button 
                onClick={() => onEdit(event)}
                className="p-2.5 text-amber-600 bg-amber-50 rounded-xl hover:bg-amber-100 transition-colors flex-1 md:flex-none flex items-center justify-center"
              >
                <Edit2 size={16} />
              </button>
              <button 
                onClick={() => onDelete(event.id)}
                className="p-2.5 text-rose-600 bg-rose-50 rounded-xl hover:bg-rose-100 transition-colors flex-1 md:flex-none flex items-center justify-center"
              >
                <Trash2 size={16} />
              </button>
            </div>
          )}
        </div>
      </div>
    </article>
  );
};

export default EventCard;
